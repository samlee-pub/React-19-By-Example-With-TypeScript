Source code:

Listing 18 contents of src/src-ts/typeScriptSpecificTypes.ts 
 1/****************************************
 2 * TypeScript-Specific Types
 3 * Build: yarn tsc --target es2021
 4 *        --module commonjs <filename>.ts
 5 * Run: node <filename>.js
 6 ***************************************/
 7//using local scope, avoiding global pollution
 8{
 9  // 1. Union Types (|)
10  let userId: string | number;
11  userId = "abc123"; // OK
12  userId = 456; // OK
13  // userId = true;  // Error (not string or number)
14  console.log(userId, ":", typeof userId);
15
16  // 2. Literal Types (Exact values)
17  let status: "active" | "inactive" | "OK" | 200 = "active"; // Literal types
18  status = "OK"; // Also allows number
19  console.log(status, ":", typeof status);
20
21  // 3. Tuples (Fixed-length arrays)
22  const person: [string, number, boolean] = ["Alice", 30, true]; // Must match structure
23  // person = [30, "Alice", true]; // Error (wrong order)
24  console.log(
25    person,
26    ":",
27    `[${typeof person[0]},${typeof person[1]},${typeof person[2]}]`,
28  );
29
30  // Named tuples (TypeScript 4.0+)
31  const coordinates: [x: number, y: number] = [10, 20];
32  console.log(
33    coordinates,
34    ":",
35    `[${typeof coordinates[0]},${typeof coordinates[1]}]`,
36  );
37
38  // 4. Any (Opt-out of type checking)
39  // eslint-disable-next-line @typescript-eslint/no-explicit-any
40  let dynamicValue: any = "Hello"; //lint error: no-explicit-any
41  dynamicValue = [1, 2, 3]; // Anything goes
42  dynamicValue = 42; // No error
43  // Use sparingly - defeats TypeScript's purpose!
44  console.log(dynamicValue, ":", typeof dynamicValue);
45
46  // 5. Unknown (Type-safe any)
47  let unknownValue: unknown = "28";
48  unknownValue = [1, 2, 3]; // Anything goes
49  unknownValue = "unknown value"; // No error
50
51  // Must type-check before use
52  if (typeof unknownValue === "string") {
53    console.log(unknownValue.toUpperCase());
54  }
55
56  // 6. Void (No return value)
57  function logMessage(message: string): void {
58    console.log(message);
59    // Implicitly returns undefined
60  }
61  console.log(logMessage, ":", typeof logMessage);
62
63  // 7. Never (Unreachable code)
64  function throwError(message: string): never {
65    throw new Error(message);
66  }
67  console.log(throwError, ":", typeof throwError);
68
69  function infiniteLoop(): never {
70    // eslint-disable-next-line no-empty
71    while (true) {} //lint error: Empty block statement
72  }
73  console.log(infiniteLoop, ":", typeof infiniteLoop);
74}

Listing 20 contents of src/src-ts/customTypesAndInterfaces.ts 
  1/****************************************
  2 * Custom Types & Interfaces
  3 * Build: yarn tsc --target es2021
  4 *        --module commonjs <filename>.ts
  5 * Run: node <filename>.js
  6 ***************************************/
  7//using local scope, avoiding global pollution
  8{
  9  // 1. Type Aliases (type)
 10  type UserID = string | number;
 11  type OddNumberUnderTen = 1 | 3 | 5 | 7 | 9 | "1" | "3" | "5" | "7" | "9"; // Union of numbers and strings
 12
 13  //Example usage of type aliases
 14  const user1Id: UserID = "user-123";
 15  const oddNumber: OddNumberUnderTen = 7;
 16  console.log("user1Id", ":", user1Id);
 17  console.log("oddNumber", ":", oddNumber);
 18
 19  // Object type
 20  type User = {
 21    id: UserID;
 22    name: string;
 23    isAdmin?: boolean; // Optional
 24  };
 25
 26  //Example usage of object type
 27  const alice: User = {
 28    id: 1,
 29    name: "Alice",
 30  };
 31  const bob: User = {
 32    id: "user-456",
 33    name: "Bob",
 34    isAdmin: true, // Optional property
 35  };
 36  console.log("Alice:", alice);
 37  console.log("Bob:", bob);
 38
 39  // 2. Interfaces (interface)
 40  // Basic interface
 41  interface IUser {
 42    id: string;
 43    name: string;
 44    email?: string; // Optional
 45  }
 46
 47  // Extending interfaces
 48  interface IAdmin extends IUser {
 49    permissions: string[];
 50  }
 51
 52  // Function interface
 53  interface MathOperation {
 54    (a: number, b: number): number;
 55  }
 56
 57  //Example usage of interfaces
 58  const user1: IUser = {
 59    id: "user1",
 60    name: "Alice",
 61  };
 62  const user2: IUser = {
 63    id: "user2",
 64    name: "Bob",
 65    email: "abc@abc.com", // Optional property
 66  };
 67  const add: MathOperation = (x, y) => x + y;
 68  console.log("User1:", user1);
 69  console.log("User2:", user2);
 70  console.log("Add 2+3:", add(2, 3));
 71
 72  // 3. Enums (Named constants)
 73  enum Direction {
 74    Up = "UP",
 75    Down = "DOWN",
 76    Left = "LEFT",
 77    Right = "RIGHT",
 78  }
 79  // Numeric enums (auto-incremented)
 80  enum StatusCode {
 81    OK = 200,
 82    NotFound = 404,
 83    ServerError = 500,
 84  }
 85
 86  // Example usage of enums
 87  const move: Direction = Direction.Up;
 88  const status: StatusCode = StatusCode.OK;
 89  console.log("Direction:", move, ":", typeof move);
 90  console.log("Status Code:", status, ":", typeof status);
 91
 92  // 4. Combining Types & Interfaces
 93  // Intersection type (&)
 94  type Name = { name: string };
 95  type Age = { age: number };
 96  type Person = Name & Age;
 97
 98  // Interface merging
 99  interface Car {
100    model: string;
101  }
102  interface Car {
103    year: number; // Merged
104  }
105  //Example usage of combined types and interfaces
106  const person: Person = { name: "Alice", age: 30 };
107  const myCar: Car = { model: "Tesla", year: 2023 };
108  console.log("Person:", person);
109  console.log("Car:", myCar);
110
111  // 5. Generic Types
112  type ApiResponse<T> = {
113    data: T;
114    error: string | null;
115  };
116
117  interface IProduct {
118    id: string;
119    price: number;
120  }
121
122  // Example usage of generic types
123  const admin: IAdmin = {
124    id: "admin1",
125    name: "Charlie",
126    permissions: ["read", "write"],
127  };
128
129  const productResponse: ApiResponse<IProduct> = {
130    data: { id: "1", price: 99 },
131    error: null,
132  };
133
134  console.log("Admin:", admin);
135  console.log("Product Response:", productResponse);
136
137  // 6. Type Assertions (as)
138  // eslint-disable-next-line @typescript-eslint/no-explicit-any
139  const pageNumber: any = "123";
140  const numericPage = pageNumber as number; // Force type
141  console.log("Numeric Page:", numericPage, ":", typeof numericPage);
142}

Listing 22 contents of src/src-ts/functionTypes.ts 
  1/****************************************
  2 * Function Types
  3 * Build: yarn tsc --target es2021
  4 *        --module commonjs <filename>.ts
  5 * Run: node <filename>.js
  6 ***************************************/
  7//using local scope, avoiding global pollution
  8{
  9  // 1. funtion declaration with type annotations
 10  function add(a: number, b: number): number {
 11    return a + b;
 12  }
 13  console.log("add(2, 3) = ", add(2, 3)); //Output: add(2, 3) =  5
 14  // 2. function expression with type annotations
 15  const multiply: (x: number, y: number) => number = function (x, y) {
 16    return x * y;
 17  };
 18  console.log("multiply(2, 3) = ", multiply(2, 3)); //Output: multiply(2, 3) =  6
 19  const multiply2 = function (x: number, y: number): number {
 20    return x * y;
 21  };
 22  console.log("multiply2(2, 3) = ", multiply2(2, 3)); //Output: multiply2(2, 3) =  6
 23  // 3. arrow function with type annotations
 24  const divide: (x: number, y: number) => number = (x, y) => {
 25    return x / y;
 26  };
 27  console.log("divide(6, 3) = ", divide(6, 3)); //Output: divide(6, 3) =  2
 28  // 4. function with optional parameters
 29  function greet(name: string, greeting?: string): string {
 30    return `${greeting || "Hello"}, ${name}!`;
 31  }
 32  console.log(greet("Alice")); //Output: Hello, Alice!
 33  console.log(greet("Bob", "Hi")); //Output: Hi, Bob!
 34  // 5. function with default parameters
 35  function sayHello(name: string, greeting: string = "Hello"): string {
 36    return `${greeting}, ${name}!`;
 37  }
 38  console.log(sayHello("Charlie")); //Output: Hello, Charlie!
 39  console.log(sayHello("Dave", "Greetings")); //Output: Greetings, Dave!
 40  // 6. function with rest parameters
 41  function sum(...numbers: number[]): number {
 42    return numbers.reduce((acc, curr) => acc + curr, 0);
 43  }
 44  console.log("sum(1, 2, 3, 4) = ", sum(1, 2, 3, 4)); //Output: sum(1, 2, 3, 4) =  10
 45  // 7. function with a callback
 46  function processNumbers(
 47    numbers: number[],
 48    callback: (num: number) => number,
 49  ): number[] {
 50    return numbers.map(callback);
 51  }
 52  const doubled = processNumbers([1, 2, 3], (num) => num * 2);
 53  console.log("Doubled numbers: ", doubled); //Output: Doubled numbers:  [2, 4, 6]
 54  // 8. function with a generic type
 55  function identity<T>(arg: T): T {
 56    return arg;
 57  }
 58  console.log("identity(5) = ", identity(5)); //Output: identity(5) =  5
 59  console.log("identity('Hello') = ", identity("Hello")); //Output: identity('Hello') =  Hello
 60  // 9. function with a type alias
 61  type StringOrNumber = string | number;
 62  function printValue(value: StringOrNumber): void {
 63    console.log("Value: ", value);
 64  }
 65  printValue("Hello"); //Output: Value:  Hello
 66  printValue(42); //Output: Value:  42
 67  // 10. function with an interface
 68  interface User {
 69    name: string;
 70    age: number;
 71  }
 72  function displayUser(user: User): void {
 73    console.log(`User: ${user.name}, Age: ${user.age}`);
 74  }
 75  displayUser({ name: "Alice", age: 30 }); //Output: User: Alice, Age: 30
 76  // 11. function with a type guard
 77  // eslint-disable-next-line @typescript-eslint/no-explicit-any
 78  function isString(value: any): value is string {
 79    return typeof value === "string";
 80  }
 81  function processValue(value: string | number): void {
 82    if (isString(value)) {
 83      console.log("String value: ", value);
 84    } else {
 85      console.log("Number value: ", value);
 86    }
 87  }
 88  processValue("Hello"); //Output: String value:  Hello
 89  processValue(42); //Output: Number value:  42
 90  //12. function with object type
 91  function displayUser2(user: { name: string; age: number }): void {
 92    console.log(`User: ${user.name}, Age: ${user.age}`);
 93  }
 94  displayUser2({ name: "Alice", age: 30 }); //Output: User: Alice, Age: 30
 95  //13. function with destructuring
 96  function displayUser3({ name, age }: { name: string; age: number }): void {
 97    console.log(`User: ${name}, Age: ${age}`);
 98  }
 99  displayUser3({ name: "Alice", age: 30 }); //Output: User: Alice, Age: 30
100}
