  1. TypeScript-Specific Types:

Source code:

Listing 9 contents of src/src-ts/tsSpecificTypes.ts
  1/****************************************
  2 * TypeScript-Specific Types Documentation
  3 *
  4 * TypeScript extends JavaScript with additional type system features:
  5 * 1. Union Types (|)       - Variables that can hold multiple types
  6 * 2. Literal Types         - Exact value constraints
  7 * 3. Tuples                - Fixed-length, typed arrays
  8 * 4. Any Type              - Opt-out of type checking (use sparingly)
  9 * 5. Unknown Type          - Type-safe alternative to 'any'
 10 * 6. Void Type             - Absence of return value
 11 * 7. Never Type            - Values that never occur
 12 *
 13 * Build: yarn tsc --target es2021 --module commonjs tsSpecificTypes.ts
 14 * Run: node tsSpecificTypes.js
 15 ***************************************/
 16
 17//using local scope, avoiding global pollution
 18{
 19  // 1. Union Types (|)
 20  let userId: string | number;
 21  userId = "abc123"; // OK - string type
 22  userId = 456; // OK - number type
 23  // userId = true;  // Error - boolean not in union type
 24  console.log("Union Type - userId:", userId, "| Type:", typeof userId);
 25
 26  // 2. Literal Types (Exact values)
 27  let status: "active" | "inactive" | "OK" | 200 = "active";
 28  status = "OK"; // Valid - "OK" is in literal union
 29  status = 200; // Valid - 200 is in literal union
 30  // status = "pending"; // Error - "pending" not in allowed literals
 31  console.log("Literal Type - status:", status, "| Type:", typeof status);
 32
 33  // 3. Tuple (Fixed-length array with specified types)
 34  const person: [string, number, boolean] = ["Alice", 30, true];
 35  // person = [30, "Alice", true]; // Error - wrong type order
 36  console.log(
 37    "Tuple - person:",
 38    person,
 39    "| Types:",
 40    `[${typeof person[0]}, ${typeof person[1]}, ${typeof person[2]}]`,
 41  );
 42
 43  // Named tuple: Tuple with labeled elements (TypeScript 4.0+)
 44  const coordinates: [x: number, y: number] = [10, 20];
 45  console.log(
 46    "Named Tuple - coordinates:",
 47    coordinates,
 48    "| Types:",
 49    `[${typeof coordinates[0]}, ${typeof coordinates[1]}]`,
 50  );
 51
 52  // 4. Any (Opt-out of type checking)
 53  // eslint-disable-next-line @typescript-eslint/no-explicit-any
 54  let dynamicValue: any = "Hello"; // Lint error: no-explicit-any
 55  dynamicValue = [1, 2, 3]; // No type checking - anything allowed
 56  dynamicValue = 42; // No error - complete type freedom
 57  console.log(
 58    "Any Type - dynamicValue:",
 59    dynamicValue,
 60    "| Type:",
 61    typeof dynamicValue,
 62  );
 63
 64  // 5. Unknown (Type-safe any)
 65  let unknownValue: unknown = "28";
 66  unknownValue = [1, 2, 3]; // Can assign anything like 'any'
 67  unknownValue = "unknown value"; // No assignment restrictions
 68
 69  // Type checking required before operations
 70  if (typeof unknownValue === "string") {
 71    console.log("Unknown Type - string value:", unknownValue.toUpperCase());
 72  } else if (Array.isArray(unknownValue)) {
 73    console.log("Unknown Type - array length:", unknownValue.length);
 74  }
 75
 76  // 6. Void (No return value)
 77  function logMessage(message: string): void {
 78    console.log(`VOID FUNCTION: ${message}`);
 79    // No return statement - implicitly returns undefined
 80  }
 81  logMessage("Hello from void function!");
 82  console.log(
 83    "Void Function - logMessage:",
 84    logMessage,
 85    "| Type:",
 86    typeof logMessage,
 87  );
 88
 89  // 7. Never (Unreachable code)
 90  function throwError(message: string): never {
 91    throw new Error(message);
 92  }
 93  console.log(
 94    "Never Type - throwError:",
 95    throwError,
 96    "| Type:",
 97    typeof throwError,
 98  );
 99
100  function infiniteLoop(): never {
101    // eslint-disable-next-line @typescript-eslint/no-unnecessary-condition
102    while (true) {
103      // Infinite loop to ensure the function never returns
104    }
105  }
106  console.log(
107    "Never Type - infiniteLoop:",
108    infiniteLoop,
109    "| Type:",
110    typeof infiniteLoop,
111  );
112}

  1/****************************************
  2 * Custom Types & Interfaces Documentation
  3 *
  4 * TypeScript provides powerful tools for defining custom types:
  5 * 1. Type Aliases (type)     - Create custom type definitions
  6 * 2. Interfaces (interface)  - Define object shapes and contracts
  7 * 3. Enums                  - Named constant collections
  8 * 4. Type Composition       - Combine types using unions and intersections
  9 * 5. Generic Types          - Create reusable type templates
 10 * 6. Type Assertions        - Override TypeScript's type inference
 11 *
 12 * Build: yarn tsc --target es2021 --module commonjs customTypes.ts
 13 * Run: node customTypes.js
 14 ***************************************/

 16// Using block scope ({}) to avoid global namespace pollution
 17{
 18  // 1. Type Aliases (type)
 19  type UserID = string | number;
 20  type OddNumberUnderTen = 1 | 3 | 5 | 7 | 9 | "1" | "3" | "5" | "7" | "9";
 21  type MathOperation = (a: number, b: number) => number; // Function type alias

 25  const oddNumber: OddNumberUnderTen = 7; // Valid - literal number

 88  // 4. Type composition - Combining Types

 90  interface Name {
 91    name: string;
 92  }

 94    age: number;
 95  }
 96  type Person = Name & Age; // Must have both name and age

100    model: string;
101  }
102
103  interface Car {
104    year: number; // Merged with previous Car declaration
105  }
106
107  // Example usage of composed types
108  const person: Person = {
109    name: "Alice",
110    age: 30, // Must satisfy both Name and Age interfaces
111  };

113  const myCar: Car = {
114    model: "Tesla",
115    year: 2023, // Must include properties from both Car declarations
116  };

118  console.log("4. Composed Intersection Type - Person:", person);
119  console.log("4. Composed Merged Interface - Car:", myCar);

121  // 5. Generic Types - Reusable Type Templates
122  interface ApiResponse<T> {
123    data: T;
124    error: string | null;
125  }
126
127  interface IProduct {
128    id: string;
129    price: number;
130  }
131
132  // Example usage of generic types and interface extension
133  const admin: IAdmin = {
134    id: "admin1",
135    name: "Charlie",
136    permissions: ["read", "write"], // Required by IAdmin interface
137  };
138
139  const productResponse: ApiResponse<IProduct> = {

141      id: "1",

145  };

151  // eslint-disable-next-line @typescript-eslint/no-explicit-any

153  const numericPage = pageNumber as number; // Type assertion

  1/****************************************
  2 * Function Types Documentation
  3 *

 13 * 9. Object Parameters - Inline object type definitions
